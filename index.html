function doGet(e) {

}
function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    if (d.action==='create') return createOrder(d);
    if (d.action==='update') return updateOrder(d);
    if (d.action==='delete') return deleteOrder(d);
    return ok('no action');
  } catch(err) { return jsonOut({error:err.message}); }
}
function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let s = ss.getSheetByName(SHEET_NAME);
  if (!s) {
    s = ss.insertSheet(SHEET_NAME);
    s.appendRow(['id','date','name','phone','wilaya','deliv','baladia','track','price','type','note','status']);
    s.getRange(1,1,1,12).setFontWeight('bold').setBackground('#1a1a2e').setFontColor('white');
  }
  return s;
}
